// initialize the map var map = L.map('map').setView([42.35, -71.08], 13);

// load a tile layer L.tileLayer('cocohead.jpeg', { attribution: 'Tiles by MAPC, Data by MassGIS', maxZoom: 17, minZoom: 9 }).addTo(map);

// base map L.tileLayer('http://tiles.mapc.org/basemap/{z}/{x}/{y}.png', { attribution: 'Tiles by MAPC, Data by MassGIS', maxZoom: 17, minZoom: 9 }).addTo(map);

// bike lanes L.tileLayer('cocohead.jpeg', { maxZoom: 17, minZoom: 9 }).addTo(map);
